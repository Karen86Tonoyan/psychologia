    <script>
        document.addEventListener('DOMContentLoaded', () => {

            const quizData = {
                quiz1: {
                    questions: [
                        "Czy często czuję, że jestem wyjątkowy/a i zasługuję na specjalne traktowanie?",
                        "Czy w rozmowach głównie czekam na swoją kolej, by mówić o sobie?",
                        "Czy trudno mi szczerze przyznać się do błędu i przeprosić?",
                        "Czy zdarza mi się wykorzystywać innych do osiągnięcia własnych celów?",
                        "Czy mam problem z odczuwaniem empatii i wczuwaniem się w sytuację innych?"
                    ],
                    interpretation: "Im więcej odpowiedzi „tak”, tym większa potrzeba przyjrzenia się swojemu ego."
                },
                quiz2: {
                    questions: [
                        "Czy często obwiniam partnera/bliskich za moje złe samopoczucie?",
                        "Czy zdarza mi się stosować ciche dni lub manipulację, aby postawić na swoim?",
                        "Czy czuję, że moje poczucie wartości zależy od aprobaty i nastroju drugiej osoby?",
                        "Czy regularnie rezygnuję z własnych potrzeb, aby uniknąć konfliktu?",
                        "Czy czuję potrzebę kontrolowania partnera/bliskich (gdzie jest, co robi)?"
                    ],
                    interpretation: "Im więcej odpowiedzi „tak”, tym większe ryzyko, że tworzysz lub tkwisz w niezdrowych relacjach."
                },
                quiz3: {
                    questions: [
                        "Czy gdy czuję stres lub smutek, moją pierwszą myślą jest sięgnięcie po używkę, jedzenie, zakupy lub telefon?",
                        "Czy często krytykuję siebie w myślach, używając obraźliwych słów?",
                        "Czy trudno mi przerwać destrukcyjne zachowanie, nawet gdy wiem, że mi szkodzi?",
                        "Czy zdarza mi się sabotować własne plany i działania, gdy zbliżam się do sukcesu?",
                        "Czy czuję wewnętrzny przymus, by być ciągle zajętym/ą i produktywnym/ą?"
                    ],
                    interpretation: "Im więcej odpowiedzi „tak”, tym ważniejsza jest praca nad zdrowymi sposobami radzenia sobie z emocjami."
                }
            };

            function createQuiz(quizId) {
                const data = quizData[quizId];
                const questionsContainer = document.getElementById(`questions${quizId.slice(-1)}`);
                
                data.questions.forEach((q, index) => {
                    const questionId = `${quizId}-q${index}`;
                    const questionElement = document.createElement('div');
                    questionElement.className = 'py-1';
                    questionElement.innerHTML = `
                        <p class="mb-1">${q}</p>
                        <div class="flex items-center space-x-4">
                            <label class="flex items-center"><input type="radio" name="${questionId}" value="tak" class="mr-1 h-4 w-4 border-gray-300 text-teal-600 focus:ring-teal-500"> Tak</label>
                            <label class="flex items-center"><input type="radio" name="${questionId}" value="nie" class="mr-1 h-4 w-4 border-gray-300 text-teal-600 focus:ring-teal-500"> Nie</label>
                        </div>
                    `;
                    questionsContainer.appendChild(questionElement);
                });
            }

            function handleSubmit(event, quizId) {
                event.preventDefault();
                const data = quizData[quizId];
                let score = 0;
                const totalQuestions = data.questions.length;
                let allAnswered = true;

                for (let i = 0; i < totalQuestions; i++) {
                    const questionId = `${quizId}-q${i}`;
                    const selected = document.querySelector(`input[name="${questionId}"]:checked`);
                    if (!selected) {
                        allAnswered = false;
                        break;
                    }
                    if (selected.value === 'tak') {
                        score++;
                    }
                }
                
                const resultsContainer = document.getElementById(`results${quizId.slice(-1)}`);
                if (!allAnswered) {
                     resultsContainer.innerHTML = `<p class="text-sm text-red-600">Proszę odpowiedzieć na wszystkie pytania.</p>`;
                     resultsContainer.classList.remove('hidden');
                     return;
                }

                const percentage = (score / totalQuestions) * 100;

                resultsContainer.innerHTML = `
                    <p class="font-semibold">Twój wynik: ${score} na ${totalQuestions} odpowiedzi "tak".</p>
                    <div class="w-full bg-stone-200 rounded-full h-2.5 mt-2">
                        <div class="bg-teal-600 h-2.5 rounded-full" style="width: ${percentage}%"></div>
                    </div>
                    <p class="text-sm text-stone-600 mt-2">${data.interpretation}</p>
                `;
                resultsContainer.classList.remove('hidden');
            }

            createQuiz('quiz1');
            createQuiz('quiz2');
            createQuiz('quiz3');

            document.getElementById('quiz1').addEventListener('submit', (e) => handleSubmit(e, 'quiz1'));
            document.getElementById('quiz2').addEventListener('submit', (e) => handleSubmit(e, 'quiz2'));
            document.getElementById('quiz3').addEventListener('submit', (e) => handleSubmit(e, 'quiz3'));
            
            const sections = document.querySelectorAll('section');
            const navLinks = document.querySelectorAll('.nav-link');

            const observerOptions = {
                root: null,
                rootMargin: '0px',
                threshold: 0.4
            };

            const observer = new IntersectionObserver((entries, observer) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        navLinks.forEach(link => {
                            link.classList.remove('active');
                            if (link.getAttribute('href').substring(1) === entry.target.id) {
                                link.classList.add('active');
                            }
                        });
                    }
                });
            }, observerOptions);

            sections.forEach(section => {
                observer.observe(section);
            });
        });
    </script>
